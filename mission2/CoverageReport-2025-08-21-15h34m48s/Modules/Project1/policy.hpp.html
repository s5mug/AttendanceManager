enum DATE_ID {
	MONDAY = 0,
	TUSEDAY = 1,
	WEDNESDAY = 2,
	THURSDAY = 3,
	FRIDAY = 4,
	SATURDAY = 5,
	SUNDAY = 6,
	MAX_NUM_DATE = 7
};


struct CountByWeekend {
	CountByWeekend() {
		memset(count_of_date, 0, sizeof(count_of_date));
	}

	int count_of_date[MAX_NUM_DATE];
};

inline bool IsValidDate(DATE_ID date_id) {
	return date_id != MAX_NUM_DATE;
}

inline DATE_ID getDateIndexFromDateStr(const string& dateStr) {
	if (dateStr == "monday") {
		return MONDAY;
	}
	if (dateStr == "tuesday") {
		return TUSEDAY;
	}
	if (dateStr == "wednesday") {
		return WEDNESDAY;
	}
	if (dateStr == "thursday") {
		return THURSDAY;
	}
	if (dateStr == "friday") {
		return FRIDAY;
	}
	if (dateStr == "saturday") {
		return SATURDAY;
	}
	if (dateStr == "sunday") {
		return SUNDAY;
	}
	return MAX_NUM_DATE;
}
